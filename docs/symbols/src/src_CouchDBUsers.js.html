  6 define("CouchDBUsers", 
  7 		
  8 ["Promise"],
  9 
 10 /**
 11  * @class
 12  * CouchDBUsers is a library with utility tools to manage users.
 13  * It helps with the creation of users and login
 14  */
 15 function CouchDBUsers(Promise) {
 16 	
 17 	return function CouchDBUsersConstructor() {
 18 		
 19 		var _transport = null;
 20 		
 21 		/**
 22 		 * Set the transport to be used, must be an Emily Transport
 23 		 * @param transport
 24 		 * @returns true if Transport is from Emily
 25 		 */
 26 		this.setTransport = function setTransport(transport) {
 27 			if (transport instanceof Object) {
 28 				_transport = transport;
 29 				return true;
 30 			} else {
 31 				return false;
 32 			}
 33 		};
 34 		
 35 		/**
 36 		 * Get the currently set transport
 37 		 * For debugging only
 38 		 * @private
 39 		 * @returns {Transport}
 40 		 */
 41 		this.getTransport = function getTransport() {
 42 			return _transport;
 43 		};
 44 		
 45 		/**
 46 		 * User login. It fetches the user document
 47 		 * the corresponds to the given name.
 48 		 * It will only work if the credentials are correct.
 49 		 * @param {String} name the name of the user
 50 		 * @param {String} password the password of the user
 51 		 * @returns {Promise}
 52 		 */
 53 		this.login = function login(name, password) {
 54 			
 55 			var promise = new Promise;
 56 			
 57 			if (typeof name == "string" && typeof password == "string") {
 58 				_transport.request("CouchDB", {
 59 					method: "GET",
 60 					path: "/_users/org.couchdb.user:"+name,
 61 					auth: name + ":" + password
 62 				}, 
 63 				promise.resolve,
 64 				promise);
 65 			} else {
 66 				promise.reject({
 67 					error: "name & password must be strings"
 68 				});
 69 			}
 70 			
 71 			return promise;
 72 			
 73 		};
 74 		
 75 		this.create = function create() {
 76 			
 77 		};
 78 		
 79 		
 80 		
 81 	};
 82 	
 83 });
