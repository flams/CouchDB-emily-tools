  6 define(["Store", "StateMachine", "Tools", "Promise"],
  7 
  8 /**
  9  * @class
 10  * CouchDBBase is a subtype of an Emily Store
 11  * and is an abstract class for CouchDBViews, BulkViews, Documents, BulkDocuments
 12  */
 13 function CouchDBBase(Store, StateMachine, Tools, Promise) {
 14 
 15 	/**
 16 	 * Duck typing.
 17 	 * @private
 18 	 */
 19 	function _isStateMachine(stateMachine) {
 20 		if (typeof stateMachine == "object" &&
 21 			typeof stateMachine.event == "function" ) {
 22 			return true;
 23 		} else {
 24 			return false;
 25 		}
 26 	}
 27 
 28 	/**
 29 	 * Double duck typing
 30 	 * @private
 31 	 */
 32 	function _isTransport(transport) {
 33 		if (typeof transport == "object" &&
 34 			typeof transport.request == "function" &&
 35 			typeof transport.listen == "function") {
 36 			return true;
 37 		} else {
 38 			return false;
 39 		}
 40 	}
 41 
 42 	/**
 43 	 * Triple duck typing
 44 	 * @private
 45 	 */
 46 	function _isPromise(promise) {
 47 		if (typeof promise == "object" &&
 48 			typeof promise.fulfill == "function" &&
 49 			typeof promise.reject == "function" &&
 50 			typeof promise.then == "function") {
 51 			return true;
 52 		} else {
 53 			return false;
 54 		}
 55 	}
 56 
 57 	/**
 58 	 * A noop function
 59 	 * @private
 60 	 */
 61 	function NOOP() {}
 62 
 63 	function CouchDBBaseConstructor() {
 64 
 65 		/**
 66 		 * It has a default state Machine
 67 		 * @private
 68 		 */
 69 		var _stateMachine = null,
 70 
 71 		/**
 72 		 * The default handler name
 73 		 * @private
 74 		 */
 75 		_handlerName = "CouchDB",
 76 
 77 		/**
 78 		 * The transport to use to issue the requests
 79 		 * @private
 80 		 */
 81 		_transport = null,
 82 
 83 		/**
 84 		 * The current synchronization informations
 85 		 * @private
 86 		 */
 87 		_syncInfo,
 88 
 89 		/**
 90 		 * A promise returned and resolved when the store is synched
 91 		 * @private
 92 		 */
 93 		_promise = new Promise;
 94 
 95 		/**
 96 		 * Set the promise to be resolved when the store is synched
 97 		 * @return {Boolean} true if it's a promise
 98 		 */
 99 		this.setPromise = function setPromise(promise) {
100 			if (_isPromise(promise)) {
101 				_promise = promise;
102 				return true;
103 			} else {
104 				return false;
105 			}
106 		};
107 
108 		/**
109 		 * Get the promise to be resolved when the store is synched
110 		 * @return {Promise} the promise
111 		 */
112 		this.getPromise = function getPromise() {
113 			return _promise;
114 		};
115 
116 		/**
117 		 * Get the current state machine
118 		 * @returns {StateMachine} the current state machine
119 		 */
120 		this.getStateMachine = function getStateMachine() {
121 			return _stateMachine;
122 		};
123 
124 		/**
125 		 * Set the state machine
126 		 * @param {StateMachine} stateMachine the state machine to set
127 		 * @returns {Boolean} true if it's an accepted state Machine
128 		 */
129 		this.setStateMachine = function setStateMachine(stateMachine) {
130 			if (_isStateMachine(stateMachine)) {
131 				_stateMachine = stateMachine;
132 				return true;
133 			} else {
134 				return false;
135 			}
136 		};
137 
138 		/**
139 		 * Get the current transport
140 		 * @returns {Transport} the current transport
141 		 */
142 		this.getTransport = function getTransport() {
143 			return _transport;
144 		};
145 
146 		/**
147 		 * Set the current transport
148 		 * @param {Transport} transport the transport to use
149 		 * @returns {Boolean} true if its an accepted transport
150 		 */
151 		this.setTransport = function setTransport(transport) {
152 			if (_isTransport(transport)) {
153 				_transport = transport;
154 				return true;
155 			} else {
156 				return false;
157 			}
158 		};
159 
160 		/**
161 		 * Get the current CouchDB handler name
162 		 * @returns {String} the current handler name
163 		 */
164 		this.getHandlerName = function getHandlerName() {
165 			return _handlerName;
166 		};
167 
168 		/**
169 		 * Set the current CouchDB handler name
170 		 * @param {String} handlerName the name of the handler
171 		 * The name must be a string that matches with the handler
172 		 * as it's been added in Emily/Olives handlers
173 		 * @returns {Boolean} true if it's a string
174 		 */
175 		this.setHandlerName = function setHandlerName(handlerName) {
176 			if (typeof handlerName == "string") {
177 				_handlerName = handlerName;
178 				return true;
179 			} else {
180 				return false;
181 			}
182 		};
183 
184 		/**
185 		 * Synchronize the store with CouchDB
186 		 * depending on the provided sync info
187 		 * @param {Object} a configuration object
188 		 * @returns {Boolean} false if no configuration object given
189 		 */
190 		this.sync = function sync() {
191 			if (_syncInfo = this.setSyncInfo.apply(this, arguments)) {
192 				_stateMachine.event("sync");
193 				return _promise;
194 			} else {
195 				return false;
196 			}
197 		};
198 
199 		/**
200 		 * Unsync the store
201 		 * @returns {Boolean} true if unsynched
202 		 */
203 		this.unsync = function unsync() {
204 			return _stateMachine.event("unsync");
205 		};
206 
207 		/**
208 		 * Returns the current synchronization info
209 		 * For debugging purpose
210 		 * @private
211 		 */
212 		this.getSyncInfo = function getSyncInfo() {
213 			return _syncInfo;
214 		};
215 
216 		/**
217 		 * This function will be called when the Store needs to be synchronized
218 		 * It's to be overriden in the sub Store
219 		 */
220 		this.onSync = function onSync() {
221 
222 		};
223 
224 		/**
225 		 * This function will be called when the Store needs to subscribe to changes
226 		 * It's to be overriden in the sub Store
227 		 */
228 		this.onListen = function onListen() {
229 
230 		};
231 
232 		/**
233 		 * This function will be called when the Store is unsynched
234 		 * It's to be overriden in the sub Store
235 		 */
236 		this.unsync = function unsync() {
237 			this.stopListening();
238 			delete this.stopListening;
239 		};
240 
241 		/**
242 		 * This function will be called when the Store needs to be subscribe to changes
243 		 * It's to be overriden in the sub Store
244 		 */
245 		this.onChange = function onChange() {
246 
247 		};
248 
249 		/**
250 		 * This function will be called when the Store needs to add something
251 		 * It's to be overriden in the sub Store
252 		 */
253 		this.onAdd = function onAdd() {
254 
255 		};
256 
257 		/**
258 		 * This function will be called when the Store needs to remove something
259 		 * It's to be overriden in the sub Store
260 		 */
261 		this.onRemove = function onRemove() {
262 
263 		};
264 
265 		/**
266 		 * This function must be overriden to validate the synchronization
267 		 * information, and set the syncInfo object.
268 		 * By default it only assigns the first arguments to syncInfo
269 		 * @params {*} arguments all the arguments
270 		 * @returns {Boolean} true
271 		 */
272 		this.setSyncInfo = function setSyncInfo(syncInfo) {
273 			return _syncInfo = syncInfo;
274 		};
275 
276 		/**
277 		 * Create the state machine with the default states
278 		 */
279 		this.setStateMachine(new StateMachine("Unsynched", {
280 
281 			"Unsynched": [
282 				["sync", this.onSync, this, "Synched"]
283 			],
284 
285 			"Synched": [
286 				["listen", this.onListen, this, "Listening"],
287 				["unsync", NOOP, "Unsynched"]
288 			],
289 
290 			"Listening": [
291 				["unsync", this.unsync, this, "Unsynched"],
292 				["change", this.onChange, this],
293 				["add", this.onAdd, this],
294 				["remove", this.onRemove, this]
295 			]
296 
297 		}));
298 
299 	}
300 
301 	return function CouchDBSBaseFactory(data) {
302 		CouchDBBaseConstructor.prototype = new Store(data);
303 		return new CouchDBBaseConstructor;
304 	};
305 
306 });
