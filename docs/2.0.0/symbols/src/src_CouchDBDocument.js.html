  6 define(["Store", "CouchDBBase", "Tools", "Promise", "StateMachine"],
  7 
  8 /**
  9  * @class
 10  * CouchDBDocument synchronizes a Store with a CouchDB document
 11  */
 12  function CouchDBDocument(Store, CouchDBBase, Tools, Promise, StateMachine) {
 13 
 14 	function CouchDBDocumentConstructor() {
 15 
 16 		/**
 17 		 * Set the synchronization data if valid data is supplied
 18 		 * @param {String} database the database to sync with
 19 		 * @param {String} document the document to request
 20 		 * @param {Object} [optional] query an object with queryparams
 21 		 * @returns {Object} syncInfo if valid, false if not
 22 		 */
 23 		 this.setSyncInfo = function setSyncInfo(database, doc, query) {
 24 			if (typeof database == "string" &&
 25 				typeof doc == "string") {
 26 
 27 			if (!query || (typeof query == "object")) {
 28 					return {
 29 						"database": database,
 30 						"document": doc,
 31 						"query": query || {}
 32 					};
 33 				} else {
 34 					return false;
 35 				}
 36 			} else {
 37 				return false;
 38 			}
 39 		 };
 40 
 41 		/**
 42 		 * Get a CouchDB document
 43 		 * @private
 44 		 */
 45 		 this.onSync = function onSync() {
 46 
 47 			var _syncInfo = this.getSyncInfo();
 48 
 49 			this.getTransport().request(
 50 				this.getHandlerName(),
 51 				{
 52 					method: "GET",
 53 					path: "/" + _syncInfo.database + "/" + _syncInfo.document,
 54 					query: _syncInfo.query
 55 				},
 56 				function (results) {
 57 					var json = JSON.parse(results);
 58 					if (json._id) {
 59 						this.reset(json);
 60 						this.getStateMachine().event("listen");
 61 					}
 62 					this.getPromise().fulfill(json);
 63 				}, this);
 64 		 };
 65 
 66 		/**
 67 		 * Subscribe to changes when synchronized with a document
 68 		 * @private
 69 		 */
 70 		 this.onListen = function onListen() {
 71 
 72 			var _syncInfo = this.getSyncInfo();
 73 
 74 			this.stopListening = this.getTransport().listen(
 75 				this.getHandlerName(),
 76 				{
 77 					path: "/" + _syncInfo.database + "/_changes",
 78 					query: {
 79 						feed: "continuous",
 80 						heartbeat: 20000,
 81 						descending: true
 82 					}
 83 				},
 84 				function (changes) {
 85 					var json;
 86 					// Should I test for this very special case (heartbeat?)
 87 					// Or do I have to try catch for any invalid json?
 88 					if (changes == "\n") {
 89 						return false;
 90 					}
 91 
 92 					json = JSON.parse(changes);
 93 
 94 					// The document is the modified document is the current one
 95 					if (json.id == _syncInfo.document &&
 96 						// And if it has a new revision
 97 						json.changes.pop().rev != this.get("_rev")) {
 98 
 99 						if (json.deleted) {
100 							this.getStateMachine().event("remove");
101 						} else {
102 							this.getStateMachine().event("change");
103 						}
104 					}
105 				}, this
106 				);
107 		 };
108 
109 		/**
110 		 * Update the document when synchronized with a document.
111 		 * @private
112 		 */
113 		 this.onChange = function onChange() {
114 
115 			var _syncInfo = this.getSyncInfo();
116 
117 			this.getTransport().request(
118 				this.getHandlerName(),
119 				{
120 					method: "GET",
121 					path: "/"+_syncInfo.database+"/" + _syncInfo.document
122 				},
123 				function (doc) {
124 					this.reset(JSON.parse(doc));
125 				}, this
126 				);
127 		 };
128 
129 		/**
130 		 * Delete all document's properties
131 		 * @private
132 		 */
133 		 this.onRemove = function onRemove() {
134 			this.reset({});
135 		 };
136 
137 		/**
138 		 * Upload the document to the database
139 		 * @returns true if synched
140 		 */
141 		 this.upload = function upload() {
142 			var promise = new Promise,
143 			_syncInfo = this.getSyncInfo();
144 
145 			if (_syncInfo.document) {
146 				this.getStateMachine().event("upload", promise);
147 				return promise;
148 			}
149 
150 			return false;
151 		 };
152 
153 		/**
154 		 * Remove the document from the database
155 		 * @returns true if remove called
156 		 */
157 		 this.remove = function remove() {
158 
159 			var _syncInfo = this.getSyncInfo(),
160 				promise = new Promise();
161 
162 			this.getStateMachine().event("removeFromDatabase", promise);
163 			return promise;
164 		 };
165 
166 		/**
167 		 * Put a new document in CouchDB
168 		 * @private
169 		 */
170 		 this.databaseCreate = function databaseCreate(promise) {
171 
172 			var _syncInfo = this.getSyncInfo();
173 
174 			this.getTransport().request(
175 				this.getHandlerName(),
176 				{
177 					method: "PUT",
178 					path: "/" + _syncInfo.database + "/" + _syncInfo.document,
179 					headers: {
180 						"Content-Type": "application/json"
181 					},
182 					data: this.toJSON()
183 				},
184 				function (result) {
185 					var json = JSON.parse(result);
186 					if (json.ok) {
187 						this.set("_rev", json.rev);
188 						this.set("_id", json.id);
189 						this.getStateMachine().event("listen");
190 						promise.fulfill(json);
191 					} else {
192 						promise.reject(json);
193 					}
194 				}, this
195 				);
196 		 };
197 
198 		/**
199 		 * Update a document in CouchDB through a PUT request
200 		 * @private
201 		 */
202 		 this.databaseUpdate = function databaseUpdate(promise) {
203 
204 			var _syncInfo = this.getSyncInfo();
205 
206 			this.getTransport().request(
207 				this.getHandlerName(),
208 				{
209 					method: "PUT",
210 					path: "/" + _syncInfo.database + "/" + _syncInfo.document,
211 					headers: {
212 						"Content-Type": "application/json"
213 					},
214 					data: this.toJSON()
215 				},
216 				function (response) {
217 					var json = JSON.parse(response);
218 					if (json.ok) {
219 						this.set("_rev", json.rev);
220 						promise.fulfill(json);
221 					} else {
222 						promise.reject(json);
223 					}
224 				},
225 				this);
226 		 };
227 
228 		/**
229 		 * Remove a document from CouchDB through a DELETE request
230 		 * @private
231 		 */
232 		 this.databaseRemove = function databaseRemove(promise) {
233 
234 			var _syncInfo = this.getSyncInfo();
235 
236 			this.getTransport().request(this.getHandlerName(),
237 			{
238 				method: "DELETE",
239 				path: "/" + _syncInfo.database + "/" + _syncInfo.document,
240 				query: {
241 					rev: this.get("_rev")
242 				}
243 			},
244 			function (response) {
245 				var json = JSON.parse(response);
246 				if (json.ok) {
247 					promise.fulfill(json);
248 				} else {
249 					promise.reject(json);
250 				}
251 			});
252 		 };
253 
254 		this.setStateMachine(new StateMachine("Unsynched", {
255 
256 			"Unsynched": [
257 				["sync", this.onSync, this, "Synched"]
258 			],
259 
260 			"Synched": [
261 				["listen", this.onListen, this, "Listening"],
262 				["unsync", function NOOP() {}, "Unsynched"],
263 				["upload", this.databaseCreate, this]
264 			],
265 
266 			"Listening": [
267 				["unsync", this.onUnsync, this, "Unsynched"],
268 				["change", this.onChange, this],
269 				["add", this.onAdd, this],
270 				["remove", this.onRemove, this],
271 				["upload", this.databaseUpdate, this],
272 				["removeFromDatabase", this.databaseRemove, this]
273 			]
274 
275 		}));
276 
277 	}
278 
279 	return function CouchDBDocumentFactory(data) {
280 		CouchDBDocumentConstructor.prototype = new CouchDBBase(data);
281 		return new CouchDBDocumentConstructor;
282 	};
283 
284 });
