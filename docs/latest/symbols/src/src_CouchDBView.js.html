  6 define(["Store", "CouchDBBase", "Tools"],
  7 
  8 /**
  9  * @class
 10  * CouchDBView synchronizes a Store with a CouchDB view
 11  */
 12 function CouchDBView(Store, CouchDBBase, Tools) {
 13 
 14 	function CouchDBViewConstructor() {
 15 
 16 		/**
 17 		 * Set the synchronization data if valid data is supplied
 18 		 * @param {String} database the database to sync with
 19 		 * @param {String} designDocument the design document to be used
 20 		 * @param {String} view the name of the view to request
 21 		 * @param {Object} [optional] query an object with queryparams
 22 		 * @returns {Object} syncInfo if valid, false if not
 23 		 */
 24 		this.setSyncInfo = function setSyncInfo(database, designDocument, view, query) {
 25 			if (typeof database == "string" &&
 26 				typeof designDocument == "string" &&
 27 				typeof view == "string") {
 28 
 29 				if (!query || (typeof query == "object")) {
 30 					return {
 31 						"database": database,
 32 						"design": designDocument,
 33 						"view": view,
 34 						"query": query || {}
 35 					};
 36 				} else {
 37 					return false;
 38 				}
 39 			} else {
 40 				return false;
 41 			}
 42 		};
 43 
 44 		/**
 45 		 * Get a CouchDB view
 46 		 * @private
 47 		 */
 48 		this.onSync = function onSync() {
 49 			var _syncInfo = this.getSyncInfo();
 50 
 51 			this.getTransport().request(this.getHandlerName(), {
 52 				method: "GET",
 53 				path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
 54 				query: _syncInfo.query
 55 			}, function (results) {
 56 				var json = JSON.parse(results);
 57 				if (!json.rows) {
 58 					throw new Error("CouchDBStore [" + _syncInfo.database + ", " + _syncInfo.design + ", " + _syncInfo.view + "].sync() failed: " + results);
 59 				} else {
 60 					this.reset(json.rows);
 61 					this.getPromise().fulfill(this);
 62 					if (typeof json.total_rows == "undefined") {
 63 						_syncInfo.reducedView = true;
 64 					}
 65 
 66 					this.getStateMachine().event("listen");
 67 				}
 68 			}, this);
 69 		};
 70 
 71 		/**
 72 		 * Subscribe to changes when synchronized with a view
 73 		 * @private
 74 		 */
 75 		this.onListen = function onListen() {
 76 
 77 			var _syncInfo = this.getSyncInfo();
 78 
 79 			Tools.mixin({
 80 				feed: "continuous",
 81 				heartbeat: 20000,
 82 				descending: true
 83 			}, _syncInfo.query);
 84 
 85 			this.stopListening = this.getTransport().listen(
 86 				this.getHandlerName(), {
 87 					path: "/" + _syncInfo.database + "/_changes",
 88 					query: _syncInfo.query
 89 				},
 90 				function (changes) {
 91 					// Should I test for this very special case (heartbeat?)
 92 					// Or do I have to try catch for any invalid json?
 93 					if (changes == "\n") {
 94 						return false;
 95 					}
 96 
 97 					var json = JSON.parse(changes),
 98 						action;
 99 
100 					// reducedView is known on the first get view
101 					if (_syncInfo.reducedView) {
102 						action = "updateReduced";
103 					} else {
104 						if (json.deleted) {
105 							action = "delete";
106 						} else if (json.changes && json.changes[0].rev.search("1-") == 0) {
107 							action = "add";
108 						} else {
109 							action = "change";
110 						}
111 					}
112 
113 					this.getStateMachine().event(action, json.id);
114 				}, this);
115 		};
116 
117 		/**
118 		 * Update in the Store a document that was updated in CouchDB
119 		 * Get the whole view :(, then get the modified document and update it.
120 		 * I have no choice but to request the whole view and look for the document
121 		 * so I can also retrieve its position in the store (idx) and update the item.
122 		 * Maybe I've missed something
123 		 * @private
124 		 */
125 		this.onChange = function onChange(id) {
126 
127 			var _syncInfo = this.getSyncInfo();
128 
129 			this.getTransport().request(
130 				this.getHandlerName(),{
131 					method: "GET",
132 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
133 					query: _syncInfo.query
134 			}, function (view) {
135 				var json = JSON.parse(view);
136 
137 				if (json.rows.length == this.getNbItems()) {
138 					json.rows.some(function (value, idx) {
139 						if (value.id == id) {
140 							this.set(idx, value);
141 						}
142 					}, this);
143 				} else {
144 					this.evenDocsInStore.call(this, json.rows, id);
145 				}
146 
147 			}, this);
148 
149 		};
150 
151 		/**
152 		 * When a doc is removed from the view even though it still exists
153 		 * or when it's added to a view, though it wasn't just created
154 		 * This function must be called to even the store
155 		 * @private
156 		 */
157 		this.evenDocsInStore = function evenDocsInStore(view, id) {
158 			var nbItems = this.getNbItems();
159 
160 			// If a document was removed from the view
161 			if (view.length < nbItems) {
162 				// Look for it in the store to remove it
163 				this.loop(function (value, idx) {
164 					if (value.id == id) {
165 						this.del(idx);
166 					}
167 				}, this);
168 
169 			// If a document was added to the view
170 			} else if (view.length > nbItems) {
171 				// Look for it in the view and add it to the store at the same place
172 				view.some(function (value, idx) {
173 					if (value.id == id) {
174 						return this.alter("splice", idx, 0, value);
175 					}
176 				}, this);
177 			}
178 
179 		};
180 
181 		/**
182 		* Add in the Store a document that was added in CouchDB
183 		* @private
184 		*/
185 		this.onAdd = function onAdd(id) {
186 
187 			var _syncInfo = this.getSyncInfo();
188 
189 			this.getTransport().request(
190 				this.getHandlerName(), {
191 					method: "GET",
192 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
193 					query: _syncInfo.query
194 			}, function (view) {
195 			var json = JSON.parse(view);
196 
197 				json.rows.some(function (value, idx) {
198 					if (value.id == id) {
199 						this.alter("splice", idx, 0, value);
200 					}
201 				}, this);
202 
203 			}, this);
204 		};
205 
206 		/**
207 		 * Remove from the Store a document that was removed in CouchDB
208 		 * @private
209 		 */
210 		this.onRemove = function onRemove(id) {
211 			this.loop(function (value, idx) {
212 				if (value.id == id) {
213 					this.del(idx);
214 				}
215 			}, this);
216 		};
217 
218 		/**
219 		 * Update a reduced view (it has one row with no id)
220 		 * @private
221 		 */
222 		this.updateReduced = function updateReduced() {
223 
224 			var _syncInfo = this.getSyncInfo();
225 
226 			this.getTransport().request(
227 				this.getHandlerName(),{
228 					method: "GET",
229 					path: "/" + _syncInfo.database + "/_design/" + _syncInfo.design + "/" + _syncInfo.view,
230 					query: _syncInfo.query
231 			}, function (view) {
232 				var json = JSON.parse(view);
233 
234 				this.set(0, json.rows[0]);
235 
236 			}, this);
237 		};
238 
239 		// Add the missing states
240 		var stateMachine = this.getStateMachine(),
241 			Listening = stateMachine.get("Listening");
242 
243 		Listening.add("updateReduced", this.updateReduced, this);
244 
245 	}
246 
247 	return function CouchDBViewFactory(data) {
248 		CouchDBViewConstructor.prototype = new CouchDBBase(data);
249 		return new CouchDBViewConstructor;
250 	};
251 
252 });
