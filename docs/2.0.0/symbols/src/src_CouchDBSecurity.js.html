  6 define(["CouchDBDocument"],
  7 
  8 /**
  9  * @class
 10  * CouchDBSecurity synchronises a CouchDBDocument with _security document
 11  */
 12 function CouchDBSecurity(CouchDBDocument) {
 13 
 14 	/**
 15 	 * Defines CouchDBSecurity
 16 	 * @returns {CouchDBSecurityConstructor}
 17 	 */
 18 	function CouchDBSecurityConstructor() {
 19 
 20 		/**
 21 		 * the name of the _security document
 22 		 * @private
 23 		 */
 24 		var _name = "_security";
 25 
 26 		/**
 27 		 * Set the name of the _security document
 28 		 * @param {String} name the name of the docuyment
 29 		 * @returns {Boolean} true if name is truthy
 30 		 */
 31 		this.setName = function setName(name) {
 32 			if (name){
 33 				_name = name;
 34 				return true;
 35 			} else {
 36 				return false;
 37 			}
 38 		};
 39 
 40 		/**
 41 		 * Get the name of the _Security document
 42 		 * @returns {String}
 43 		 */
 44 		this.getName = function getName() {
 45 			return _name;
 46 		};
 47 
 48 		/**
 49 		 * Load the security document
 50 		 * @param {String} db the name of the database
 51 		 */
 52 		this.load = function load(db) {
 53 			return this.sync(db, _name);
 54 		};
 55 
 56 
 57 	};
 58 
 59 	return function CouchDBSecurityFactory() {
 60 		CouchDBSecurityConstructor.prototype = new CouchDBDocument;
 61 		return new CouchDBSecurityConstructor;
 62 	};
 63 
 64 
 65 
 66 });
