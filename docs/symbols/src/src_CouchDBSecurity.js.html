  6 define("CouchDBSecurity", 
  7 		
  8 ["CouchDBStore"], 
  9 
 10 /**
 11  * @class
 12  * CouchDBSecurity synchronises a CouchDBStore with _security document
 13  */
 14 function CouchDBSecurity(CouchDBStore) {
 15 	
 16 	/**
 17 	 * Defines CouchDBSecurity
 18 	 * @returns {CouchDBSecurityConstructor}
 19 	 */
 20 	function CouchDBSecurityConstructor() {
 21 		
 22 		/**
 23 		 * the name of the _security document
 24 		 * @private
 25 		 */
 26 		var _name = "_security";
 27 		
 28 		/**
 29 		 * Set the name of the _security document
 30 		 * @param {String} name the name of the docuyment
 31 		 * @returns {Boolean} true if name is truthy
 32 		 */
 33 		this.setName = function setName(name) {
 34 			if (name){
 35 				_name = name;
 36 				return true;
 37 			} else {
 38 				return false;
 39 			}
 40 		};
 41 		
 42 		/**
 43 		 * Get the name of the _Security document
 44 		 * @returns {String}
 45 		 */
 46 		this.getName = function getName() {
 47 			return _name;
 48 		};
 49 		
 50 		/**
 51 		 * Load the security document
 52 		 * @param {String} db the name of the database
 53 		 */
 54 		this.load = function load(db) {
 55 			return this.sync(db, _name);
 56 		};
 57 		
 58 		
 59 	};
 60 	
 61 	return function CouchDBSecurityFactory() {
 62 		CouchDBSecurityConstructor.prototype = new CouchDBStore;
 63 		return new CouchDBSecurityConstructor;
 64 	};
 65 	
 66 	
 67 	
 68 });
