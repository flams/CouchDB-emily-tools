  6 define("CouchDBUser", 
  7 		
  8 ["CouchDBStore"], 
  9 
 10 /**
 11  * @class
 12  * CouchDBUser synchronises a CouchDBStore with a CouchDB User.
 13  * It also provides tools to ease the creation/modification of users.
 14  */
 15 function CouchDBUser(CouchDBStore) {
 16 	
 17 	/**
 18 	 * Defines CouchDBUser
 19 	 * @returns {CouchDBUserConstructor}
 20 	 */
 21 	function CouchDBUserConstructor() {
 22 		
 23 		/**
 24 		 * the name of the table in which users are saved
 25 		 * @private
 26 		 */
 27 		var _userDB = "_users",
 28 		
 29 		/**
 30 		 * the string which prefixes a user's id 
 31 		 * @private
 32 		 */
 33 		_idPrefix = "org.couchdb.user:";
 34 		
 35 		/**
 36 		 * Get the name of the users' db
 37 		 * @returns {String}
 38 		 */
 39 		this.getUserDB = function getUserDB() {
 40 			return _userDB;
 41 		};
 42 		
 43 		/**
 44 		 * Set the name of the users' db
 45 		 * @param {String} name of the db
 46 		 * @returns {Boolean} true if name truthy
 47 		 */
 48 		this.setUserDB = function setUserDB(name) {
 49 			if (name) {
 50 				_userDB = name;
 51 				return true;
 52 			} else {
 53 				return false;
 54 			}
 55 		};
 56 		
 57 		/**
 58 		 * Get the string that prefixes the users' id
 59 		 * @returns {String}
 60 		 */
 61 		this.getIdPrefix = function getIdPrefix() {
 62 			return _idPrefix;
 63 		};
 64 		
 65 		/**
 66 		 * Set the string that prefixes the users' id
 67 		 * @param {String} name string that prefixes the users' id
 68 		 * @returns {Boolean} true if name truthy
 69 		 */
 70 		this.setIdPrefix = function setIdPrefix(name) {
 71 			if (name) {
 72 				_idPrefix = name;
 73 				return true;
 74 			} else {
 75 				return false;
 76 			}
 77 		};
 78 		
 79 		/**
 80 		 * Set user's id
 81 		 * @param {String} id
 82 		 * @returns {Boolean} true if id truthy
 83 		 */
 84 		this.setId = function setId(id) {
 85 			if (id) {
 86 				this.set("_id", _idPrefix + id);
 87 				return true;
 88 			} else {
 89 				return false;
 90 			}
 91 		};
 92 		
 93 		/**
 94 		 * Get the user's id
 95 		 * @returns {String} the user's id
 96 		 */
 97 		this.getId = function getId() {
 98 			return this.get("_id");
 99 		};
100 		
101 		/**
102 		 * Load a user given it's id
103 		 * @param {String} id of the user (without prefix)
104 		 * @returns {Boolean} true if sync succeeded
105 		 */
106 		this.load = function load(id) {
107 			if (id) {
108 				return this.sync(_userDB, _idPrefix + id);
109 			} else {
110 				return false;
111 			}
112 		};
113 		
114 	};
115 	
116 	return function CouchDBUserFactory() {
117 		CouchDBUserConstructor.prototype = new CouchDBStore;
118 		return new CouchDBUserConstructor;
119 	};
120 	
121 	
122 	
123 });
